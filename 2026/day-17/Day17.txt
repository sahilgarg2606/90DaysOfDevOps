#!/bin/bash


packages=(curl nginx)

for pack in "${packages[@]}"
do
        if dpkg -s "$pack" > /dev/null 2>&1
        then
                echo "$pack is already there"
        else
                sudo apt install -y "$pack"

        fi
        if systemctl list-unit-files | grep -q "$pack"
        then
                systemctl status "$pack"
        fi
done
